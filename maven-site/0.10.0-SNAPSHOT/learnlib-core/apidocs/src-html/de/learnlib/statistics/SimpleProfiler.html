018package de.learnlib.statistics;
019
020import java.util.HashMap;
021import java.util.Map;
022import java.util.Map.Entry;
023
024import javax.annotation.Nonnull;
025import javax.annotation.ParametersAreNonnullByDefault;
026
027import de.learnlib.logging.LearnLogger;
028
029/**
030 * Very rudimentary profiler. 
031 */
032@ParametersAreNonnullByDefault
033public class SimpleProfiler {
034
035  private static boolean PROFILE = true;
036  
037  private static final Map<String,Counter> cumulated = new HashMap<>();
038  private static final Map<String,Long> pending = new HashMap<>();
039   
040  private static LearnLogger logger = LearnLogger.getLogger(SimpleProfiler.class.getName());
041  
042  /**
043   * reset internal data.
044   */
045  public static void reset() {
046      cumulated.clear();
047      pending.clear();
048  }
049  
050  /**
051   * start activity.
052   * 
053   * @param name 
054   */
055  public static void start(String name) {
056    if (!PROFILE) {
057      return;
058    }
059    long start = System.currentTimeMillis();
060  
061    pending.put(name,start);
062    
063  }
064  
065  /**
066   * stop activity.
067   * 
068   * @param name 
069   */
070  public static void stop(String name) {
071    if (!PROFILE) {
072      return;
073    }
074    Long start = pending.remove(name);
075    if (start == null) {
076      return;
077    }
078    long duration = System.currentTimeMillis() - start;
079    Counter sum = cumulated.get(name);
080    if (sum == null) {
081      sum = new Counter(name, "ms");
082    }
083    sum.increment(duration);
084    cumulated.put(name, sum);
085  }
086  
087  /**
088   * get profiling results as string.
089   * 
090   * @return 
091   */
092  @Nonnull
093  public static String getResults() {
094    StringBuilder sb = new StringBuilder();
095    for (Entry<String, Counter> e : cumulated.entrySet()) {
096        sb.append(e.getValue().getSummary()).append(", (").append(e.getValue().getCount()/1000.0).
097                append(" s)").append(System.lineSeparator());
098    }
099    return sb.toString();
100  }
101  
102  /**
103   * log results in category PROFILING.
104   */
105  public static void logResults() {
106    for (Entry<String, Counter> e : cumulated.entrySet()) {
107      logger.logProfilingInfo(e.getValue());
108    }  
109  }
110  
111}
