020import java.util.Collections;

023import net.automatalib.automata.concepts.SuffixOutput;
024import net.automatalib.automata.fsa.DFA;

027import net.automatalib.words.Word;
028
029import com.github.misberner.buildergen.annotations.GenerateBuilder;
030
031import de.learnlib.algorithms.features.observationtable.OTLearner.OTLearnerDFA;
032import de.learnlib.algorithms.lstargeneric.ExtensibleAutomatonLStar;
033import de.learnlib.algorithms.lstargeneric.ce.ObservationTableCEXHandler;
034import de.learnlib.algorithms.lstargeneric.closing.ClosingStrategy;
035import de.learnlib.algorithms.lstargeneric.table.Row;

037
038
039/**
040 * An implementation of Angluin's L* algorithm for learning DFAs, as described in the paper
041 * "Learning Regular Sets from Queries and Counterexamples".
042 * 
043 * @author Malte Isberner <malte.isberner@gmail.com>
044 *
045 * @param <I> input symbol class.
046 */
047public class ExtensibleLStarDFA<I>
048        extends ExtensibleAutomatonLStar<DFA<?,I>, I, Boolean, Integer, Integer, Boolean, Void, CompactDFA<I>>
049        implements OTLearnerDFA<I> {
050
051
052        public static class BuilderDefaults extends ExtensibleAutomatonLStar.BuilderDefaults {
053                public static <I> Collection<? extends Word<I>> initialSuffixes() {
054                        return Collections.singleton(Word.<I>epsilon());
055                }
056        }
057        
058        /**
059         * Constructor.
060         * @param alphabet the learning alphabet.
061         * @param oracle the DFA oracle.
062         */
063        @GenerateBuilder(defaults = ExtensibleAutomatonLStar.BuilderDefaults.class)
064        public ExtensibleLStarDFA(Alphabet<I> alphabet, MembershipOracle<I,Boolean> oracle,
065                        Collection<? extends Word<I>> initialSuffixes,
066                        ObservationTableCEXHandler<? super I, ? super Boolean> cexHandler,
067                        ClosingStrategy<? super I, ? super Boolean> closingStrategy) {
068                super(alphabet, oracle, new CompactDFA<I>(alphabet),
069                                LStarDFAUtil.ensureSuffixCompliancy(initialSuffixes),
070                                cexHandler, closingStrategy);
071        }
072
073
074
075        /*
076         * (non-Javadoc)
077         * @see de.learnlib.lstar.AbstractAutomatonLStar#stateProperty(de.learnlib.lstar.Row)
078         */
079        @Override
080        protected Boolean stateProperty(Row<I> stateRow) {
081                return table.cellContents(stateRow, 0);
082        }
083
084        /*
085         * (non-Javadoc)
086         * @see de.learnlib.lstar.AbstractAutomatonLStar#transitionProperty(de.learnlib.lstar.Row, int)
087         */
088        @Override
089        protected Void transitionProperty(Row<I> stateRow, int inputIdx) {
090                return null;
091        }
092
093
094        /*
095         * (non-Javadoc)
096         * @see de.learnlib.algorithms.lstargeneric.AbstractAutomatonLStar#exposeInternalHypothesis()
097         */
098        @Override
099        protected DFA<?, I> exposeInternalHypothesis() {
100                return internalHyp;
101        }
102
103        /*
104         * (non-Javadoc)
105         * @see de.learnlib.algorithms.lstargeneric.ExtensibleAutomatonLStar#hypothesisOutput()
106         */
107        @Override
108        protected SuffixOutput<I, Boolean> hypothesisOutput() {
109                return internalHyp;
110        }
111
112
113}
