017package de.learnlib.counterexamples;
018
019import java.util.Objects;
020
021import net.automatalib.automata.concepts.SuffixOutput;
022import net.automatalib.words.Word;
023import de.learnlib.api.AccessSequenceTransformer;
024import de.learnlib.api.MembershipOracle;
025import de.learnlib.api.Query;
026import de.learnlib.oracles.MQUtil;
027
028/**
029 * A collection of suffix-based local counterexample analyzers.
030 * 
031 * @see LocalSuffixFinder
032 * 
033 * @author Malte Isberner <malte.isberner@gmail.com>
034 */
035public abstract class LocalSuffixFinders {
036        
037        /**
038         * Searches for a distinguishing suffixes by checking for counterexample yielding
039         * access sequence transformations in linear ascending order.
040         * @see #findLinear(Query, AccessSequenceTransformer, SuffixOutput, MembershipOracle)
041         */
042        public static final LocalSuffixFinder<Object,Object> FIND_LINEAR
043                = new LocalSuffixFinder<Object,Object>() {
044                        @Override
045                        public <RI,RO>
046                        int findSuffixIndex(Query<RI, RO> ceQuery,
047                                        AccessSequenceTransformer<RI> asTransformer,
048                                        SuffixOutput<RI,RO> hypOutput,
049                                        MembershipOracle<RI, RO> oracle) {
050                                return findLinear(ceQuery, asTransformer, hypOutput, oracle);
051                        }
052                        @Override
053                        public String toString() {
054                                return "FindLinear";
055                        }
056        };
057        
058        /**
059         * Searches for a distinguishing suffixes by checking for counterexample yielding
060         * access sequence transformations in linear descending order.
061         * @see #findLinearReverse(Query, AccessSequenceTransformer, SuffixOutput, MembershipOracle)
062         */
063        public static final LocalSuffixFinder<Object,Object> FIND_LINEAR_REVERSE
064                = new LocalSuffixFinder<Object,Object>() {
065                        @Override
066                        public <RI,RO>
067                        int findSuffixIndex(Query<RI, RO> ceQuery,
068                                        AccessSequenceTransformer<RI> asTransformer,
069                                        SuffixOutput<RI,RO> hypOutput,
070                                        MembershipOracle<RI, RO> oracle) {
071                                return findLinearReverse(ceQuery, asTransformer, hypOutput, oracle);
072                        }
073                        @Override
074                        public String toString() {
075                                return "FindLinear-Reverse";
076                        }
077        };
078        
079        /**
080         * Searches for a distinguishing suffixes by checking for counterexample yielding
081         * access sequence transformations using a binary search, as proposed by Rivest & Schapire.
082         * @see #findRivestSchapire(Query, AccessSequenceTransformer, SuffixOutput, MembershipOracle)
083         */
084        public static final LocalSuffixFinder<Object,Object> RIVEST_SCHAPIRE
085                = new LocalSuffixFinder<Object,Object>() {
086                        @Override
087                        public <RI,RO>
088                        int findSuffixIndex(Query<RI, RO> ceQuery,
089                                        AccessSequenceTransformer<RI> asTransformer,
090                                        SuffixOutput<RI,RO> hypOutput,
091                                        MembershipOracle<RI, RO> oracle) {
092                                return findRivestSchapire(ceQuery, asTransformer, hypOutput, oracle);
093                        }
094                        @Override
095                        public String toString() {
096                                return "RivestSchapire";
097                        }
098        };
099
100        
101        
102        /**
103         * Searches for a distinguishing suffixes by checking for counterexample yielding
104         * access sequence transformations in linear ascending order.
105         * 
106         * @param ceQuery the initial counterexample query
107         * @param asTransformer the access sequence transformer
108         * @param hypOutput interface to the hypothesis output, for checking whether the oracle output
109         * contradicts the hypothesis
110         * @param oracle interface to the SUL
111         * @return the index of the respective suffix, or <tt>-1</tt> if no
112         * counterexample could be found
113         * @see LocalSuffixFinder
114         */
115        public static <S,I,O> int findLinear(Query<I,O> ceQuery,
116                        AccessSequenceTransformer<I> asTransformer,
117                        SuffixOutput<I,O> hypOutput,
118                        MembershipOracle<I, O> oracle) {
119                
120                Word<I> queryWord = ceQuery.getInput();
121                int queryLen = queryWord.length();
122                
123                Word<I> prefix = ceQuery.getPrefix();
124                int prefixLen = prefix.length();
125                
126                // If the prefix is an access sequence (i.e., a short prefix),
127                // then we can omit the first step, as transforming won't change
128                int min = asTransformer.isAccessSequence(prefix) ? prefixLen+1 : prefixLen;
129                
130                for(int i = min; i <= queryLen; i++) {
131                        Word<I> nextPrefix = queryWord.prefix(i);
132                        Word<I> as = asTransformer.transformAccessSequence(nextPrefix);
133                        
134                        Word<I> nextSuffix = queryWord.subWord(i);
135                        
136                        O hypOut = hypOutput.computeSuffixOutput(as, nextSuffix);
137                        O mqOut = MQUtil.output(oracle, as, nextSuffix);
138                        
139                        if(Objects.equals(hypOut, mqOut))
140                                return i;
141                }
142                
143                return -1;
144        }
145        
146        /**
147         * Searches for a distinguishing suffixes by checking for counterexample yielding
148         * access sequence transformations in linear descending order.
149         * 
150         * @param ceQuery the initial counterexample query
151         * @param asTransformer the access sequence transformer
152         * @param hypOutput interface to the hypothesis output, for checking whether the oracle output
153         * contradicts the hypothesis
154         * @param oracle interface to the SUL
155         * @return the index of the respective suffix, or <tt>-1</tt> if no
156         * counterexample could be found
157         * @see LocalSuffixFinder
158         */
159        public static <I,O> int findLinearReverse(Query<I,O> ceQuery,
160                        AccessSequenceTransformer<I> asTransformer,
161                        SuffixOutput<I,O> hypOutput,
162                        MembershipOracle<I, O> oracle) {
163                
164                Word<I> queryWord = ceQuery.getInput();
165                int queryLen = queryWord.length();
166                
167                Word<I> prefix = ceQuery.getPrefix();
168                int prefixLen = prefix.length();
169                
170                // If the prefix is no access sequence (i.e., a long prefix),
171                // then we also need to consider that breakage only occurs
172                // by transforming this long prefix into a short one
173                int min = asTransformer.isAccessSequence(prefix) ? prefixLen : prefixLen-1;
174                
175                for(int i = queryLen - 1; i >= min; i--) {
176                        Word<I> nextPrefix = queryWord.prefix(i);
177                        Word<I> as = asTransformer.transformAccessSequence(nextPrefix);
178                        Word<I> nextSuffix = queryWord.subWord(i);
179                        
180                        O hypOut = hypOutput.computeSuffixOutput(as, nextSuffix);
181                        O mqOut = MQUtil.output(oracle, as, nextSuffix);
182                        
183                        if(!Objects.equals(hypOut, mqOut))
184                                return i+1;
185                }
186                
187                return -1;
188        }
189        
190        
191        /**
192         * Searches for a distinguishing suffixes by checking for counterexample yielding
193         * access sequence transformations using a binary search, as proposed by Rivest & Schapire.
194         * 
195         * @param ceQuery the initial counterexample query
196         * @param asTransformer the access sequence transformer
197         * @param hypOutput interface to the hypothesis output, for checking whether the oracle output
198         * contradicts the hypothesis
199         * @param oracle interface to the SUL
200         * @return the index of the respective suffix, or <tt>-1</tt> if no
201         * counterexample could be found
202         * @see LocalSuffixFinder
203         */
204        public static <I,O> int findRivestSchapire(Query<I,O> ceQuery,
205                        AccessSequenceTransformer<I> asTransformer,
206                        SuffixOutput<I,O> hypOutput,
207                        MembershipOracle<I, O> oracle) {
208
209                Word<I> queryWord = ceQuery.getInput();
210                int queryLen = queryWord.length();
211                
212                Word<I> prefix = ceQuery.getPrefix();
213                int prefixLen = prefix.length();
214                
215                
216                int low = asTransformer.isAccessSequence(prefix) ? prefixLen : prefixLen-1;
217                
218                int high = queryLen;
219                
220                while((high - low) > 1) {
221                        int mid = low + (high - low + 1)/2;
222                        
223                        
224                        Word<I> nextPrefix = queryWord.prefix(mid);
225                        Word<I> as = asTransformer.transformAccessSequence(nextPrefix);
226                        
227                        Word<I> nextSuffix = queryWord.subWord(mid);
228                        
229                        O hypOut = hypOutput.computeSuffixOutput(as, nextSuffix);
230                        O ceOut = MQUtil.output(oracle, as, nextSuffix);
231                        
232                        if(!Objects.equals(hypOut, ceOut))
233                                low = mid;
234                        else
235                                high = mid;
236                }
237                
238                // FIXME: No check if actually found CE
239                return low+1;
240        }
241        
242        @SuppressWarnings("unchecked")
243        public static LocalSuffixFinder<Object, Object>[] values() {
244                return new LocalSuffixFinder[]{
245                                FIND_LINEAR,
246                                FIND_LINEAR_REVERSE,
247                                RIVEST_SCHAPIRE
248                };
249        }
250        
251        // Prevent inheritance
252        private LocalSuffixFinders() {}
