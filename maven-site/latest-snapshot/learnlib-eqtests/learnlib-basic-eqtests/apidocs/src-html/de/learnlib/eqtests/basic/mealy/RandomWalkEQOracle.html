017package de.learnlib.eqtests.basic.mealy;
018
019import java.util.Collection;
020import java.util.List;
021import java.util.Random;
022import java.util.logging.Logger;
023
024import net.automatalib.automata.transout.MealyMachine;
025import net.automatalib.commons.util.collections.CollectionsUtil;
026import net.automatalib.words.Word;
027import net.automatalib.words.WordBuilder;
028import de.learnlib.api.EquivalenceOracle.MealyEquivalenceOracle;
029import de.learnlib.api.SUL;
030import de.learnlib.oracles.DefaultQuery;
031
032/**
033 * Performs a random walk over the hypothesis. A random walk restarts with a
034 * fixed probability after every step and terminates after a fixed number of
035 * steps or with a counterexample. The number of steps to termination may be
036 * reset for every new search.
037 * 
038 * @param <A>
039 *            hypothesis format
040 * @param <I>
041 *            input symbols class
042 * @param <O>
043 *            output symbol class
044 * 
045 * @author falkhowar
046 */
047public class RandomWalkEQOracle<I, O>
048                implements MealyEquivalenceOracle<I,O> {
049        
050        private static final Logger LOGGER = Logger.getLogger(RandomWalkEQOracle.class.getName());
051
052        /**
053         * probability to restart before step.
054         */
055        private final double restartProbability;
056
057        /**
058         * maximum number of steps.
059         */
060        private final long maxSteps;
061
062        /**
063         * step counter.
064         */
065        private long steps = 0;
066
067        /**
068         * flag for reseting step count after every search.
069         */
070        private boolean resetStepCount;
071
072        /**
073         * RNG.
074         */
075        private final Random random;
076
077        /**
078         * System under learning.
079         */
080        private final SUL<I, O> sul;
081
082        /**
083         * Constructor.
084         * 
085         * @param restartProbability
086         * @param maxSteps
087         * @param random
088         * @param sul
089         */
090        public RandomWalkEQOracle(double restartProbability, long maxSteps,
091                        Random random, SUL<I, O> sul) {
092                this.restartProbability = restartProbability;
093                this.maxSteps = maxSteps;
094                this.random = random;
095                this.sul = sul;
096        }
097
098        public RandomWalkEQOracle(double restartProbability, long maxSteps,
099                        boolean resetStepCount, Random random, SUL<I, O> sul) {
100                this(restartProbability, maxSteps, random, sul);
101                this.resetStepCount = resetStepCount;
102        }
103
104        /**
105         * 
106         * @param hypothesis
107         * @param inputs
108         * @return null or a counterexample
109         */
110        @Override
111        public DefaultQuery<I, Word<O>> findCounterExample(MealyMachine<?,I,?,O> hypothesis,
112                        Collection<? extends I> inputs) {
113                return doFindCounterExample(hypothesis, inputs);
114        }
115
116        private <S, T> DefaultQuery<I, Word<O>> doFindCounterExample(
117                        MealyMachine<S, I, T, O> hypothesis, Collection<? extends I> inputs) {
118                // reset termination counter?
119                if (resetStepCount) {
120                        steps = 0;
121                }
122                
123                if(inputs.isEmpty()) {
124                        LOGGER.warning("Passed empty set of inputs to equivalence oracle; no counterexample can be found!");
125                        return null;
126                }
127
128                List<? extends I> choices = CollectionsUtil.randomAccessList(inputs);
129                int bound = choices.size();
130                S cur = hypothesis.getInitialState();
131                WordBuilder<I> wbIn = new WordBuilder<>();
132                WordBuilder<O> wbOut = new WordBuilder<>();
133
134                boolean first = true;
135                sul.pre();
136                while (steps < maxSteps) {
137
138                        // restart?
139                        double restart = random.nextDouble();
140                        if (restart < restartProbability) {
141                                if (first) {
142                                    first = false;
143                                }
144                                else {
145                                    sul.post();
146                                }
147                                sul.pre();
148                                cur = hypothesis.getInitialState();
149                                wbIn.clear();
150                                wbOut.clear();
151                        }
152
153                        // step
154                        steps++;
155                        I in = choices.get(random.nextInt(bound));
156                        O outSul = sul.step(in);
157                        T hypTrans = hypothesis.getTransition(cur, in);
158                        O outHyp = hypothesis.getTransitionOutput(hypTrans);
159                        wbIn.add(in);
160                        wbOut.add(outSul);
161
162                        // ce?
163                        if (!outSul.equals(outHyp)) {
164                                DefaultQuery<I, Word<O>> ce = new DefaultQuery<>(wbIn.toWord());
165                                ce.answer(wbOut.toWord());
166                                sul.post();
167                                return ce;
168                        }
169                        cur = hypothesis.getSuccessor(cur, in);
170                }
171                sul.post();
172                return null;
173        }
174}
