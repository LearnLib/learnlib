018package de.learnlib.examples.mealy;
019
020import java.io.IOException;
021import java.io.Writer;
022import net.automatalib.automata.transout.impl.FastMealy;
023import net.automatalib.automata.transout.impl.FastMealyState;
024import net.automatalib.commons.dotutil.DOT;
025import net.automatalib.util.graphs.dot.GraphDOT;
026import net.automatalib.words.Alphabet;
027import net.automatalib.words.impl.FastAlphabet;
028import net.automatalib.words.impl.Symbol;
029
030/**
031 * This example represents the Coffee Machine example from
032 * Steffen et al. "Introduction to Active Automata Learning from
033 * a Practical Perspective" (Figure 3)
034 * 
035 * @author Maik Merten <maikmerten@googlemail.com>
036 */
037public class ExampleCoffeeMachine {
038        
039        public final static Symbol in_water = new Symbol("water");
040        public final static Symbol in_pod = new Symbol("pod");
041        public final static Symbol in_button = new Symbol("button");
042        public final static Symbol in_clean = new Symbol("clean");
043        
044        public final static String out_ok = "ok";
045        public final static String out_error = "error";
046        public final static String out_coffee = "coffee!";
047        
048        public final static Alphabet<Symbol> alphabet = new FastAlphabet<>(in_water, in_pod, in_button, in_clean);
049        
050    /**
051     * Construct and return a machine representation of this example
052     * 
053     * @return a Mealy machine representing the coffee machine example
054     */   
055    public static FastMealy<Symbol, String> constructMachine() {
056
057        FastMealy<Symbol, String> fm = new FastMealy<>(alphabet);
058                
059                FastMealyState<String> a = fm.addInitialState(),
060                                b = fm.addState(),
061                                c = fm.addState(),
062                                d = fm.addState(),
063                                e = fm.addState(),
064                                f = fm.addState();
065                
066                fm.addTransition(a, in_water, c, out_ok);
067                fm.addTransition(a, in_pod, b, out_ok);
068                fm.addTransition(a, in_button, f, out_error);
069                fm.addTransition(a, in_clean, a, out_ok);
070
071                fm.addTransition(b, in_water, d, out_ok);
072                fm.addTransition(b, in_pod, b, out_ok);
073                fm.addTransition(b, in_button, f, out_error);
074                fm.addTransition(b, in_clean, a, out_ok);
075
076                fm.addTransition(c, in_water, c, out_ok);
077                fm.addTransition(c, in_pod, d, out_ok);
078                fm.addTransition(c, in_button, f, out_error);
079                fm.addTransition(c, in_clean, a, out_ok);
080
081                fm.addTransition(d, in_water, d, out_ok);
082                fm.addTransition(d, in_pod, d, out_ok);
083                fm.addTransition(d, in_button, e, out_coffee);
084                fm.addTransition(d, in_clean, a, out_ok);
085
086                fm.addTransition(e, in_water, f, out_error);
087                fm.addTransition(e, in_pod, f, out_error);
088                fm.addTransition(e, in_button, f, out_error);
089                fm.addTransition(e, in_clean, a, out_ok);
090                
091                fm.addTransition(f, in_water, f, out_error);
092                fm.addTransition(f, in_pod, f, out_error);
093                fm.addTransition(f, in_button, f, out_error);
094                fm.addTransition(f, in_clean, f, out_error);
095
096        return fm;
097    }
098
099    public static void main(String[] args) throws IOException {
100
101        FastMealy<Symbol, String> fm = constructMachine();
102
103        Writer w = DOT.createDotWriter(true);
104        GraphDOT.write(fm, fm.getInputAlphabet(), w);
105        w.close();
106    }
107
108        
109}
