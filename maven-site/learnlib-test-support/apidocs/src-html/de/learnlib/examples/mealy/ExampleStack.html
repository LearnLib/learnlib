018package de.learnlib.examples.mealy;
019
020import net.automatalib.automata.transout.impl.FastMealy;
021import net.automatalib.automata.transout.impl.FastMealyState;
022import net.automatalib.words.Alphabet;
023import net.automatalib.words.impl.FastAlphabet;
024import net.automatalib.words.impl.Symbol;
025
026/**
027 * This example encodes a small stack with a capacity of three elements
028 * and "push" and "pop" operations as Mealy machine. Outputs are
029 * "ok", "empty" and "full".
030 * 
031 * @author Maik Merten <maikmerten@googlemail.com>
032 */
033public class ExampleStack {
034    private final static Symbol in_push = new Symbol("push");
035    private final static Symbol in_pop = new Symbol("pop");
036    private final static Alphabet<Symbol> alphabet = new FastAlphabet<>(in_push, in_pop); 
037    
038    private final static String out_ok = "ok";
039    private final static String out_empty = "empty";
040    private final static String out_full = "full";
041    
042    
043    /**
044     * Construct and return a machine representation of this example
045     * 
046     * @return machine instance of the example
047     */
048    public static FastMealy<Symbol, String> constructMachine() {
049
050        FastMealy<Symbol, String> fm = new FastMealy<>(alphabet);
051        
052        FastMealyState<String> s0 = fm.addInitialState(),
053                s1 = fm.addState(),
054                s2 = fm.addState(),
055                s3 = fm.addState();
056        
057        fm.addTransition(s0, in_push, s1, out_ok);
058        fm.addTransition(s0, in_pop, s0, out_empty);
059        
060        fm.addTransition(s1, in_push, s2, out_ok);
061        fm.addTransition(s1, in_pop, s0, out_ok);
062        
063        fm.addTransition(s2, in_push, s3, out_ok);
064        fm.addTransition(s2, in_pop, s1, out_ok);
065        
066        fm.addTransition(s3, in_push, s3, out_full);
067        fm.addTransition(s3, in_pop, s2, out_ok);
068        
069        return fm;
070    }
071    
072}
