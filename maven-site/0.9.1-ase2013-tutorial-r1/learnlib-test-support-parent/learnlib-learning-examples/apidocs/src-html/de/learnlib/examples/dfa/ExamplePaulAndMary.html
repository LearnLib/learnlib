018package de.learnlib.examples.dfa;
019
020import net.automatalib.automata.fsa.MutableDFA;
021import net.automatalib.automata.fsa.impl.compact.CompactDFA;
022import net.automatalib.util.automata.builders.AutomatonBuilders;
023import net.automatalib.words.Alphabet;
024import net.automatalib.words.impl.FastAlphabet;
025import net.automatalib.words.impl.Symbol;
026import de.learnlib.examples.DefaultLearningExample.DefaultDFALearningExample;
027
028/**
029 * This class implements a sad love story - DFA style.
030 * 
031 * @author Maik Merten <maikmerten@googlemail.com>
032 */
033public class ExamplePaulAndMary extends DefaultDFALearningExample<Symbol> {
034        
035
036        public static final Symbol IN_PAUL = new Symbol("Paul");
037    public static final Symbol IN_LOVES = new Symbol("loves");
038    public static final Symbol IN_MARY = new Symbol("Mary");
039    
040 
041    
042    public static Alphabet<Symbol> createInputAlphabet() {
043        return new FastAlphabet<>(IN_PAUL, IN_LOVES, IN_MARY);
044    }
045  
046    
047    /**
048     * Construct and return a machine representation of this example
049     * 
050     * @return machine instance of the example
051     */
052    public static CompactDFA<Symbol> constructMachine() {
053        return constructMachine(new CompactDFA<>(createInputAlphabet()));
054    }
055     
056    public static <A extends MutableDFA<S,? super Symbol>,S>
057    A constructMachine(A dfa) {
058
059        /*
060                S s0 = dfa.addInitialState(false), s1 = dfa.addState(false), s2 = dfa
061                                .addState(false), s3 = dfa.addState(true), s4 = dfa
062                                .addState(false);
063
064                dfa.addTransition(s0, IN_PAUL, s1);
065                dfa.addTransition(s0, IN_LOVES, s4);
066                dfa.addTransition(s0, IN_MARY, s4);
067
068                dfa.addTransition(s1, IN_PAUL, s4);
069                dfa.addTransition(s1, IN_LOVES, s2);
070                dfa.addTransition(s1, IN_MARY, s4);
071
072                dfa.addTransition(s2, IN_PAUL, s4);
073                dfa.addTransition(s2, IN_LOVES, s4);
074                dfa.addTransition(s2, IN_MARY, s3);
075
076                dfa.addTransition(s3, IN_PAUL, s4);
077                dfa.addTransition(s3, IN_LOVES, s4);
078                dfa.addTransition(s3, IN_MARY, s4);
079
080                dfa.addTransition(s4, IN_PAUL, s4);
081                dfa.addTransition(s4, IN_LOVES, s4);
082                dfa.addTransition(s4, IN_MARY, s4);
083                */
084
085        dfa = AutomatonBuilders.forDFA(dfa)
086                        .withInitial("s0")
087                        .from("s0")
088                                .on(IN_PAUL).to("s1")
089                                .on(IN_LOVES, IN_MARY).to("s4")
090                        .from("s1")
091                                .on(IN_LOVES).to("s2")
092                                .on(IN_PAUL, IN_MARY).to("s4")
093                        .from("s2")
094                                .on(IN_MARY).to("s3")
095                                .on(IN_PAUL, IN_LOVES).to("s4")
096                        .from("s3")
097                                .on(IN_PAUL, IN_LOVES, IN_MARY).to("s4")
098                        .from("s4")
099                                .on(IN_PAUL, IN_LOVES, IN_MARY).loop()
100                        .withAccepting("s3")
101                .create();
102        
103        return dfa;
104    }
105    
106    public static ExamplePaulAndMary createExample() {
107        return new ExamplePaulAndMary();
108    }
109    
110    public ExamplePaulAndMary() {
111                super(constructMachine());
112        }
113
114}
