040/**
041 * This example shows how to use the reuse filter on the
042 * {@link BoundedStringQueue} of {@link de.learnlib.examples.example2.Example}.
043 * 
044 * Please note that there is no equivalence oracle used in this example so the
045 * resulting mealy machines are only first "guesses".
046 * 
047 * @author Oliver Bauer <oliver.bauer@tu-dortmund.de>
048 */
049public class Example {
050        public static void main(String[] args) throws Exception {
051                Example example = new Example();
052                System.out.println("--");
053                System.out.println("Run experiment 1 (ReuseOracle):");
054                MealyMachine<?, String, ?, String> result1 = example.runExperiment1();
055                System.out.println("--");
056                System.out.println("Run experiment 2:");
057                MealyMachine<?, String, ?, String> result2 = example.runExperiment2();
058                System.out.println("--");
059                System.out.println("Model 1: " + result1.size() + " states");
060                System.out.println("Model 2: " + result2.size() + " states");
061
062                Word<String> sepWord = null;
063                sepWord = Automata.findSeparatingWord(result1, result2, example.sigma);
064
065                System.out.println("Difference (separating word)? " + sepWord);
066
067                /*
068                 * Background knowledge: L^* creates an observation table with
069                 * |S|=3,|SA|=7,|E|=3 so 30 MQs should be sufficient. The reuse filter
070                 * should sink system states from S*E to SA*E so the upper part is the
071                 * number of saved resets (=9).
072                 * 
073                 * If the numbers don't match:
074                 * https://github.com/LearnLib/learnlib/issues/5 (9 queries will be
075                 * pumped by the reuse filter)
076                 */
077        }
078
079        private String offer1 = "offer_1";
080        private String offer2 = "offer_2";
081        private String poll = "poll";
082        private Alphabet<String> sigma;
083        private List<Word<String>> initialSuffixes;
084
085        public Example() {
086                sigma = new SimpleAlphabet<>();
087                sigma.add(offer1);
088                sigma.add(offer2);
089                sigma.add(poll);
090
091                initialSuffixes = new ArrayList<>();
092                for (String symbol : sigma) {
093                        initialSuffixes.add(Word.fromLetter(symbol));
094                }
095        }
096
097        /*
098         * A "normal" scenario without reuse filter technique.
099         */
100        public MealyMachine<?, String, ?, String> runExperiment1() throws Exception {
101                // For each membership query a new instance of BoundedStringQueue will
102                // be created (normal learning scenario without filters)
103                FullMembershipQueryOracle oracle = new FullMembershipQueryOracle();
104
105                // construct L* instance (almost classic Mealy version)
106                // almost: we use words (Word<String>) in cells of the table
107                // instead of single outputs.
108                MealyLearner<String, String> lstar = null;
109                lstar = new ExtensibleLStarMealyBuilder<String, String>()
110                                .withAlphabet(sigma).withInitialSuffixes(initialSuffixes)
111                                .withOracle(oracle).create();
112
113                lstar.startLearning();
114                MealyMachine<?, String, ?, String> result = null;
115                result = lstar.getHypothesisModel();
116
117                System.out.println("Resets:  " + oracle.resets);
118                System.out.println("Symbols: " + oracle.symbols);
119
120                return result;
121        }
122
123        /*
124         * Scenario with reuse filter technique.
125         */
126        public MealyMachine<?, String, ?, String> runExperiment2() throws Exception {
127                MySystemStateHandler ssh = new MySystemStateHandler();
128
129                // This time we use the reuse filter to avoid some resets and
130                // save execution of symbols
131                ReuseCapableImplFactory factory = new ReuseCapableImplFactory();
132                ReuseOracle<BoundedStringQueue, String, String> reuseOracle;
133                reuseOracle = new ReuseOracleBuilder<>(sigma, factory)
134                                .withSystemStateHandler(ssh)
135                                .build();
136
137                // construct L* instance (almost classic Mealy version)
138                // almost: we use words (Word<String>) in cells of the table
139                // instead of single outputs.
140
141                MealyLearner<String, String> lstar = null;
142                lstar = new ExtensibleLStarMealyBuilder<String, String>()
143                                .withAlphabet(sigma)
144                                .withInitialSuffixes(initialSuffixes)
145                                .withOracle(reuseOracle).create();
146
147                lstar.startLearning();
148
149                // get learned model
150                MealyMachine<?, String, ?, String> result = lstar.getHypothesisModel();
151
152                // now invalidate all system states and count the number of disposed
153                // queues (equals number of resets)
154                reuseOracle.getReuseTree().disposeSystemstates();
155                ReuseCapableImpl reuseCapableOracle = (ReuseCapableImpl) reuseOracle.getReuseCapableOracle();
156                System.out.println("Resets:   " + reuseCapableOracle.fullQueries);
157                System.out.println("Reused:   " + reuseCapableOracle.reused);
158                System.out.println("Symbols:  " + reuseCapableOracle.symbols);
159                // disposed = resets
160                System.out.println("Disposed: " + ssh.disposed);
161
162                return result;
163        }
164
165        /**
166         * For running the example this class could also be removed/ignored. It is
167         * only here for documentation purposes.
168         */
169        class MySystemStateHandler implements
170                        SystemStateHandler<BoundedStringQueue> {
171                private int disposed = 0;
172
173                @Override
174                public void dispose(BoundedStringQueue state) {
175                        /*
176                         * When learning e.g. examples that involve databases, here would be
177                         * a good point to remove database entities. In this example we just
178                         * count the number of disposed (garbage collection inside the reuse
179                         * tree) objects.
180                         */
181                        disposed++;
182                }
183        }
184
185        /**
186         * An oracle that also does the reset by creating a new instance of the
187         * {@link BoundedStringQueue}.
188         */
189        class FullMembershipQueryOracle implements
190                        MealyMembershipOracle<String, String> {
191                private int resets = 0;
192                private int symbols = 0;
193
194                @Override
195                public void processQueries(
196                                Collection<? extends Query<String, Word<String>>> queries) {
197                        for (Query<String, Word<String>> query : queries) {
198                                resets++;
199                                symbols += query.getInput().size();
200
201                                BoundedStringQueue s = new BoundedStringQueue();
202
203                                WordBuilder<String> output = new WordBuilder<>();
204                                for (String input : query.getInput()) {
205                                        output.add(exec(s, input));
206                                }
207
208                                query.answer(output.toWord().suffix(query.getSuffix().size()));
209                        }
210                }
211        }
212
213        class ReuseCapableImplFactory implements Supplier<ReuseCapableOracle<BoundedStringQueue, String, String>> {
214
215                @Override
216                public ReuseCapableOracle<BoundedStringQueue, String, String> get() {
217                        return new ReuseCapableImpl();
218                }
219        }
220
221        /**
222         * An implementation of the {@link ReuseCapableOracle} needed for the
223         * {@link ReuseOracle}. It only does resets (by means of creating a new
224         * {@link BoundedStringQueue} instance in
225         * {@link ReuseCapableOracle#processQuery(Word)}.
226         */
227        class ReuseCapableImpl implements
228                        ReuseCapableOracle<BoundedStringQueue, String, String> {
229
230                private int reused = 0;
231                private int fullQueries = 0;
232                private int symbols = 0;
233
234                @Override
235                public QueryResult<BoundedStringQueue, String> continueQuery(
236                                Word<String> trace, BoundedStringQueue s) {
237
238                        reused++;
239                        symbols += trace.size();
240
241                        WordBuilder<String> output = new WordBuilder<>();
242
243                        for (String input : trace) {
244                                output.add(exec(s, input));
245                        }
246
247                        QueryResult<BoundedStringQueue, String> result;
248                        result = new QueryResult<>(output.toWord(), s);
249
250                        return result;
251                }
252
253                @Override
254                public QueryResult<BoundedStringQueue, String> processQuery(
255                                Word<String> trace) {
256                        fullQueries++;
257                        symbols += trace.size();
258
259                        /* Suppose the reset would be a time consuming operation */
260                        BoundedStringQueue s = new BoundedStringQueue();
261
262                        WordBuilder<String> output = new WordBuilder<>();
263
264                        for (String input : trace) {
265                                output.add(exec(s, input));
266                        }
267
268                        QueryResult<BoundedStringQueue, String> result;
269                        result = new QueryResult<>(output.toWord(), s);
270
271                        return result;
272                }
273        }
274
275        private String exec(BoundedStringQueue s, String input) {
276                if (input.equals(offer1) || input.equals(offer2)) {
277                        s.offer(input);
278                        return "void";
279                } else if (input.equals(poll)) {
280                        return s.poll();
281                } else {
282                        throw new RuntimeException("unknown input symbol");
283                }
284        }
