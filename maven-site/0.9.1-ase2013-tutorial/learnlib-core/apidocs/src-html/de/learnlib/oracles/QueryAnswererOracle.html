017package de.learnlib.oracles;
018
019import java.util.Collection;
020
021import de.learnlib.api.MembershipOracle;
022import de.learnlib.api.Query;
023import de.learnlib.api.QueryAnswerer;
024
025public class QueryAnswererOracle<I, O> implements MembershipOracle<I,O> {
026        
027        private final QueryAnswerer<I,O> answerer;
028
029        public QueryAnswererOracle(QueryAnswerer<I,O> answerer) {
030                this.answerer = answerer;
031        }
032
033        @Override
034        public void processQueries(Collection<? extends Query<I, O>> queries) {
035                MQUtil.answerQueries(answerer, queries);
036        }
037        
038}
