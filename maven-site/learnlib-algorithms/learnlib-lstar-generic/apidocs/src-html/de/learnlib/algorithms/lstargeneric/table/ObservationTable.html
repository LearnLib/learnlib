033/**
034 * Observation table class.
035 * <p>
036 * An observation table (OT) is the central data structure used by Angluin's L* algorithm,
037 * as described in the paper "Learning Regular Sets from Queries and Counterexamples".
038 * <p>
039 * An observation table is a two-dimensional table, with rows indexed by prefixes,
040 * and columns indexed by suffixes. For a prefix <code>u</code> and a suffix <code>v</code>,
041 * the respective cell contains the result of the membership query <code>(u, v)</code>.
042 * <p>
043 * The set of prefixes (row labels) is divided into two disjoint sets: short and long prefixes.
044 * Each long prefix is a one-letter extension of a short prefix; conversely, every time a
045 * prefix is added to the set of short prefixes, all possible one-letter extensions are added
046 * to the set of long prefixes.
047 * <p>
048 * In order to derive a well-defined hypothesis from an observation table, it must satisfy two
049 * properties: closedness and consistency.
050 * <ul>
051 * <li>An observation table is <b>closed</b> iff for each long prefix <code>u</code> there exists
052 * a short prefix <code>u'</code> such that the row contents for both prefixes are equal.
053 * <li>An observation table is <b>consistent</b> iff for every two short prefixes <code>u</code> and
054 * <code>u'</code> with identical row contents, it holds that for every input symbol <code>a</code>
055 * the rows indexed by <code>ua</code> and <code>u'a</code> also have identical contents.  
056 * </ul>
057 * 
058 * @author Malte Isberner <malte.isberner@gmail.com>
059 *
060 * @param <I> input symbol class
061 * @param <O> output class
062 */
063public final class ObservationTable<I,O> implements AccessSequenceTransformer<I> {
064        
065

086                = new HashMap<Word<I>,Row<I>>();

102        /**
103         * Initializes an observation table using a specified set of suffixes.
104         * 
105         * @param initialSuffixes the set of initial column labels.
106         * @param oracle the {@link MembershipOracle} to use for performing queries
107         * @return a list of equivalence classes of unclosed rows
108         */
109        public List<List<Row<I>>> initialize(List<Word<I>> initialSuffixes, MembershipOracle<I,O> oracle) {
110                if(allRows.size() > 0)
111                        throw new IllegalStateException("Called initialize, but there are already rows present");
112                
113                int numSuffixes = initialSuffixes.size();
114                suffixes.addAll(initialSuffixes);
115                
116                int numLps = alphabet.size();
117                int numPrefixes = 1 + numLps;
118                
119                List<DefaultQuery<I,O>> queries = new ArrayList<DefaultQuery<I,O>>(numPrefixes * numSuffixes);
120                
121                Word<I> eps = Word.epsilon();
122                Row<I> epsRow = createSpRow(Word.<I>epsilon());
123                
124                buildQueries(queries, eps, suffixes);
125                
126                for(int i = 0; i < alphabet.size(); i++) {
127                        I sym = alphabet.getSymbol(i);
128                        Word<I> w = Word.fromLetter(sym);
129                        Row<I> lpRow = createLpRow(w);
130                        buildQueries(queries, w, suffixes);
131                        epsRow.setSuccessor(i, lpRow);
132                }
133                
134                oracle.processQueries(queries);
135                
136                Iterator<DefaultQuery<I,O>> queryIt = queries.iterator();
137                
138                List<O> firstRowContents = new ArrayList<O>(numSuffixes);
139                fetchResults(queryIt, firstRowContents, numSuffixes);
140                processContents(epsRow, firstRowContents, true);
141                
142                List<List<Row<I>>> unclosed = new ArrayList<List<Row<I>>>();
143                
144                for(Row<I> lpRow : longPrefixRows) {
145                        List<O> rowContents = new ArrayList<O>(numSuffixes);
146                        fetchResults(queryIt, rowContents, numSuffixes);
147                        if(processContents(lpRow, rowContents, false))
148                                unclosed.add(new ArrayList<Row<I>>());
149                        
150                        int id = lpRow.getRowContentId();
151                        
152                        if(id > 0)
153                                unclosed.get(id - 1).add(lpRow);
154                }
155                
156                return unclosed;
157        }

159        /**
160         * Adds a suffix to the list of distinguishing suffixes. This is a convenience method
161         * that can be used as shorthand for
162         * <code>addSufixes(Collections.singletonList(suffix), oracle)</code>
163         * @param suffix the suffix to add
164         * @param oracle the membership oracle
165         * @return a list of equivalence classes of unclosed rows
166         */
167        public List<List<Row<I>>> addSuffix(Word<I> suffix, MembershipOracle<I,O> oracle) {
168                return addSuffixes(Collections.singletonList(suffix), oracle);
169        }
170        
171        /**
172         * Adds suffixes to the list of distinguishing suffixes.
173         * @param newSuffixes the suffixes to add
174         * @param oracle the membership oracle
175         * @return a list of equivalence classes of unclosed rows
176         */
177        public List<List<Row<I>>> addSuffixes(List<Word<I>> newSuffixes, MembershipOracle<I, O> oracle) {
178                int oldSuffixCount = suffixes.size();
179                int numNewSuffixes = newSuffixes.size();
180                
181                int numSpRows = shortPrefixRows.size();
182                int rowCount = numSpRows + longPrefixRows.size();
183                
184                List<DefaultQuery<I,O>> queries = new ArrayList<DefaultQuery<I,O>>(rowCount * numNewSuffixes);
185                
186                for(Row<I> row : shortPrefixRows)
187                        buildQueries(queries, row.getPrefix(), newSuffixes);
188                
189                for(Row<I> row : longPrefixRows)
190                        buildQueries(queries, row.getPrefix(), newSuffixes);
191                
192                oracle.processQueries(queries);
193                
194                Iterator<DefaultQuery<I,O>> queryIt = queries.iterator();
195                
196                for(Row<I> row : shortPrefixRows) {
197                        List<O> rowContents = allRowContents.get(row.getRowContentId());
198                        if(rowContents.size() == oldSuffixCount) {
199                                rowContentIds.remove(rowContents);
200                                fetchResults(queryIt, rowContents, numNewSuffixes);
201                                rowContentIds.put(rowContents, row.getRowContentId());
202                        }
203                        else {
204                                List<O> newContents = new ArrayList<O>(oldSuffixCount + numNewSuffixes);
205                                newContents.addAll(rowContents.subList(0, oldSuffixCount));
206                                fetchResults(queryIt, newContents, numNewSuffixes);
207                                processContents(row, newContents, true);
208                        }
209                }
210                
211                List<List<Row<I>>> unclosed = new ArrayList<List<Row<I>>>();
212                numSpRows = numDistinctRows();
213                
214                for(Row<I> row : longPrefixRows) {
215                        List<O> rowContents = allRowContents.get(row.getRowContentId());
216                        if(rowContents.size() == oldSuffixCount) {
217                                rowContentIds.remove(rowContents);
218                                fetchResults(queryIt, rowContents, numNewSuffixes);
219                                rowContentIds.put(rowContents, row.getRowContentId());
220                        }
221                        else {
222                                List<O> newContents = new ArrayList<O>(oldSuffixCount + numNewSuffixes);
223                                newContents.addAll(rowContents.subList(0, oldSuffixCount));
224                                fetchResults(queryIt, newContents, numNewSuffixes);
225                                if(processContents(row, newContents, false))
226                                        unclosed.add(new ArrayList<Row<I>>());
227                                
228                                int id = row.getRowContentId();
229                                if(id >= numSpRows)
230                                        unclosed.get(id - numSpRows).add(row);
231                        }       
232                }
233                
234                this.suffixes.addAll(newSuffixes);
235                
236                return unclosed;
237        }

240        /**
241         * Moves the specified rows to the set of short prefix rows. If some of the specified
242         * rows already are short prefix rows, they are ignored (unless they do not have any
243         * contents, in which case they are completed).
244         * @param lpRows the rows to move to the set of short prefix rows
245         * @param oracle the membership oracle
246         * @return a list of equivalence classes of unclosed rows
247         */
248        public List<List<Row<I>>> toShortPrefixes(List<Row<I>> lpRows, MembershipOracle<I,O> oracle) {
249                List<Row<I>> freshSpRows = new ArrayList<Row<I>>();
250                List<Row<I>> freshLpRows = new ArrayList<Row<I>>();
251                
252                for(Row<I> row : lpRows) {
253                        if(row.isShortPrefix()) {
254                                if(row.hasContents())
255                                        continue;
256                                freshSpRows.add(row);
257                        }
258                        else {
259                                makeShort(row);
260                                if(!row.hasContents())
261                                        freshSpRows.add(row);
262                        }
263                        
264                        Word<I> prefix = row.getPrefix();
265                                                
266                        for(int i = 0; i < alphabet.size(); i++) {
267                                I sym = alphabet.getSymbol(i);
268                                Word<I> lp = prefix.append(sym);
269                                Row<I> lpRow = rowMap.get(lp);
270                                if(lpRow == null) {
271                                        lpRow = createLpRow(lp);
272                                        freshLpRows.add(lpRow);
273                                }
274                                row.setSuccessor(i, lpRow);
275                        }
276                }
277                
278                int numSuffixes = suffixes.size();
279                
280                int numFreshRows = freshSpRows.size() + freshLpRows.size();
281                List<DefaultQuery<I,O>> queries = new ArrayList<DefaultQuery<I,O>>(numFreshRows * numSuffixes);
282                buildRowQueries(queries, freshSpRows, suffixes);
283                buildRowQueries(queries, freshLpRows, suffixes);
284                
285                oracle.processQueries(queries);
286                Iterator<DefaultQuery<I,O>> queryIt = queries.iterator();
287                
288                for(Row<I> row : freshSpRows) {
289                        List<O> contents = new ArrayList<O>(numSuffixes);
290                        fetchResults(queryIt, contents, numSuffixes);
291                        processContents(row, contents, true);
292                }
293                
294                int numSpRows = numDistinctRows();
295                List<List<Row<I>>> unclosed = new ArrayList<List<Row<I>>>();
296                
297                for(Row<I> row : freshLpRows) {
298                        List<O> contents = new ArrayList<O>(numSuffixes);
299                        fetchResults(queryIt, contents, numSuffixes);
300                        if(processContents(row, contents, false))
301                                unclosed.add(new ArrayList<Row<I>>());
302                        
303                        int id = row.getRowContentId();
304                        if(id >= numSpRows)
305                                unclosed.get(id - numSpRows).add(row);
306                }
307                
308                return unclosed;
309        }

311        public Row<I> getRowSuccessor(Row<I> row, I sym) {
312                return row.getSuccessor(alphabet.getSymbolIndex(sym));
313        }
314        
315        public List<List<Row<I>>> addShortPrefixes(List<Word<I>> shortPrefixes, MembershipOracle<I,O> oracle) {     
316                List<Row<I>> toSpRows = new ArrayList<Row<I>>();
317                
318                for(Word<I> sp : shortPrefixes) {
319                        Row<I> row = rowMap.get(sp);
320                        if(row != null) {
321                                if(row.isShortPrefix())
322                                        continue;
323                        }
324                        else
325                                row = createSpRow(sp);
326                        toSpRows.add(row);
327                }
328                
329                return toShortPrefixes(toSpRows, oracle);
330        }

333        @SuppressWarnings("unchecked")
334        public Inconsistency<I,O> findInconsistency() {
335                Row<I>[] canonicRows = (Row<I>[])new Row<?>[numDistinctRows()];
336                
337                for(Row<I> spRow : shortPrefixRows) {
338                        int contentId = spRow.getRowContentId();
339                        
340                        Row<I> canRow = canonicRows[contentId];
341                        if(canRow == null) {
342                                canonicRows[contentId] = spRow;
343                                continue;
344                        }
345                        
346                        for(int i = 0; i < alphabet.size(); i++) {
347                                int spSuccContent = spRow.getSuccessor(i).getRowContentId();
348                                int canSuccContent = canRow.getSuccessor(i).getRowContentId();
349                                if(spSuccContent != canSuccContent)
350                                        return new Inconsistency<I,O>(canRow, spRow, i);
351                        }
352                }
353                
354                return null;
355        }

358        protected boolean makeShort(Row<I> row) {
359                if(row.isShortPrefix())
360                        return false;
361                
362                int lastIdx = longPrefixRows.size() - 1;
363                Row<I> last = longPrefixRows.get(lastIdx);
364                int rowIdx = row.getLpIndex();
365                longPrefixRows.remove(lastIdx);
366                if(last != row) {
367                        longPrefixRows.set(rowIdx, last);
368                        last.setLpIndex(rowIdx);
369                }
370                
371                shortPrefixRows.add(row);
372                row.makeShort(alphabet.size());
373                
374                if(row.hasContents()) {
375                        int cid = row.getRowContentId();
376                        if(canonicalRows.get(cid) == null)
377                                canonicalRows.set(cid, row);
378                }
379                return true;
380        }
381        
382        protected Row<I> createLpRow(Word<I> prefix) {
383                Row<I> newRow = new Row<I>(prefix, numRows++);
384                allRows.add(newRow);
385                rowMap.put(prefix, newRow);
386                int idx = longPrefixRows.size();
387                longPrefixRows.add(newRow);
388                newRow.setLpIndex(idx);
389                return newRow;
390        }
391        
392        protected Row<I> createSpRow(Word<I> prefix) {
393                Row<I> newRow = new Row<I>(prefix, numRows++, alphabet.size());
394                allRows.add(newRow);
395                rowMap.put(prefix, newRow);
396                shortPrefixRows.add(newRow);
397                return newRow;
398        }

441        protected boolean processContents(Row<I> row, List<O> rowContents, boolean makeCanonical) {
442                Integer contentId = rowContentIds.get(rowContents);
443                boolean added = false;
444                if(contentId == null) {
445                        rowContentIds.put(rowContents, contentId = numDistinctRows());
446                        allRowContents.add(rowContents);
447                        added = true;
448                        if(makeCanonical)
449                                canonicalRows.add(row);
450                        else
451                                canonicalRows.add(null);
452                }
453                row.setRowContentId(contentId);
454                return added;
455        }

457        protected static <I,O>
458        void buildQueries(List<DefaultQuery<I,O>> queryList, List<Word<I>> prefixes, List<Word<I>> suffixes) {
459                for(Word<I> prefix : prefixes)
460                        buildQueries(queryList, prefix, suffixes);
461        }
462        
463        protected static <I,O>
464        void buildRowQueries(List<DefaultQuery<I,O>> queryList, List<Row<I>> rows, List<Word<I>> suffixes) {
465                for(Row<I> row : rows)
466                        buildQueries(queryList, row.getPrefix(), suffixes);
467        }
468        
469        protected static <I,O>
470        void buildQueries(List<DefaultQuery<I,O>> queryList, Word<I> prefix, List<Word<I>> suffixes) {
471                for(Word<I> suffix : suffixes)
472                        queryList.add(new DefaultQuery<I,O>(prefix, suffix));
473        }

475        /**
476         * Fetches the given number of query responses and adds them to the specified output list.
477         * Also, the query iterator is advanced accordingly.
478         * @param queryIt the query iterator
479         * @param output the output list to write to
480         * @param numSuffixes the number of suffixes (queries)
481         */
482        protected static <I,O>
483        void fetchResults(Iterator<DefaultQuery<I,O>> queryIt, List<O> output, int numSuffixes) {
484                for(int j = 0; j < numSuffixes; j++) {
485                        DefaultQuery<I,O> qry = queryIt.next();
486                        output.add(qry.getOutput());
487                }
488        }

506        /*
507         * (non-Javadoc)
508         * @see de.learnlib.api.AccessSequenceTransformer#transformAccessSequence(net.automatalib.words.Word)
509         */
510        @Override
511        public Word<I> transformAccessSequence(Word<I> word) {
512                Row<I> current = shortPrefixRows.get(0);
513                
514                for(I sym : word) {
515                        current = getRowSuccessor(current, sym);
516                        if(!current.isShortPrefix())
517                                current = canonicalRows.get(current.getRowContentId());
518                }
519                
520                return current.getPrefix();
521        }

523        /*
524         * (non-Javadoc)
525         * @see de.learnlib.api.AccessSequenceTransformer#isAccessSequence(net.automatalib.words.Word)
526         */
527        @Override
528        public boolean isAccessSequence(Word<I> word) {
529                Row<I> current = shortPrefixRows.get(0);
530                
531                for(I sym : word) {
532                        current = getRowSuccessor(current, sym);
533                        if(!current.isShortPrefix())
534                                return false;
535                }
536                
537                return true;
538        }
